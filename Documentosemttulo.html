<script>

    const form = document.getElementById('contactForm');

    const basicTableBody = document.getElementById('basicContactTableBody');

    const notification = document.getElementById('notification');

    const contacts = JSON.parse(localStorage.getItem('contacts')) || [];

    let editIndex = -1;

    function showForm() {

        form.reset();

        document.getElementById('photoPreview').style.display = 'none';

        document.getElementById('formContainer').style.display = 'block';

    }

    function hideForm() {

        document.getElementById('formContainer').style.display = 'none';

        editIndex = -1;

    }

    function notifyExpiredContacts() {

        const today = new Date();

        const expiredContacts = contacts.filter(contact => {

            const lastContactDate = new Date(contact.lastContact);

            return lastContactDate && (today - lastContactDate) > 15 * 24 * 60 * 60 * 1000;

        });

        notification.style.display = expiredContacts.length > 0 ? 'block' : 'none';

    }

    function previewPhoto() {

        const file = document.getElementById('photo').files[0];

        if (file) {

            const reader = new FileReader();

            reader.onload = function (e) {

                document.getElementById('photoPreview').src = e.target.result;

                document.getElementById('photoPreview').style.display = 'block';

            };

            reader.readAsDataURL(file);

        }

    }

    form.addEventListener('submit', (e) => {

        e.preventDefault();

        const contact = {

            name: document.getElementById('name').value,

            phone: document.getElementById('phone').value,

            dob: document.getElementById('dob').value,

            city: document.getElementById('city').value,

            group: document.getElementById('group').value,

            priority: document.getElementById('priority').value,

            lastContact: document.getElementById('lastContact').value,

            notes: document.getElementById('notes').value,

            photo: document.getElementById('photo').files[0] ? URL.createObjectURL(document.getElementById('photo').files[0]) : ''

        };

        if (editIndex === -1) {

            contacts.push(contact);

        } else {

            contacts[editIndex] = contact;

        }

        localStorage.setItem('contacts', JSON.stringify(contacts));

        renderContacts();

        hideForm();

        notifyExpiredContacts();

    });

    function renderContacts() {

        basicTableBody.innerHTML = '';

        contacts.forEach((contact, index) => {

            const isAttention = new Date() - new Date(contact.lastContact) > 15 * 24 * 60 * 60 * 1000;

            const tr = document.createElement('tr');

            if (isAttention) {

                tr.classList.add('attention');

            }

            tr.innerHTML = `

                <td><img src="${contact.photo}" alt="Foto" class="contact-photo"></td>

                <td><a href="#" onclick="toggleDetails(${index})">${contact.name}</a></td>

                <td><a href="https://www.google.com/maps/search/${encodeURIComponent(contact.city)}" target="_blank">${contact.city}</a></td>

                <td>

                    <button class="btn btn-warning btn-sm" onclick="editContact(${index})">Editar</button>

                    <button class="btn btn-danger btn-sm" onclick="deleteContact(${index})">Excluir</button>

                </td>

            `;

            const detailsRow = document.createElement('tr');

            detailsRow.classList.add('details-row');

            detailsRow.id = `details-${index}`;

            detailsRow.innerHTML = `

                <td colspan="4">

                    <strong>Telefone:</strong> <a href="https://wa.me/${contact.phone.replace(/\D/g, '')}" target="_blank">${contact.phone}</a><br>

                    <strong>Data de Nascimento:</strong> ${contact.dob}<br>

                    <strong>Grupo:</strong> ${contact.group}<br>

                    <strong>Prioridade:</strong> ${contact.priority}<br>

                    <strong>Último Contato:</strong> ${contact.lastContact}<br>

                    <strong>Notas:</strong> ${contact.notes}<br>

                </td>

            `;

            basicTableBody.appendChild(tr);

            basicTableBody.appendChild(detailsRow);

        });

    }

    function toggleDetails(index) {

        const row = document.getElementById(`details-${index}`);

        row.style.display = row.style.display === 'none' ? 'table-row' : 'none';

    }

    function editContact(index) {

        const contact = contacts[index];

        editIndex = index;

        document.getElementById('name').value = contact.name;

        document.getElementById('phone').value = contact.phone;

        document.getElementById('dob').value = contact.dob;

        document.getElementById('city').value = contact.city;

        document.getElementById('group').value = contact.group;

        document.getElementById('priority').value = contact.priority;

        document.getElementById('lastContact').value = contact.lastContact;

        document.getElementById('notes').value = contact.notes;

        document.getElementById('photo').value = '';

        document.getElementById('photoPreview').style.display = 'none';

        showForm();

    }

    function deleteContact(index) {

        if (confirm('Tem certeza que deseja excluir este contato?')) {

            contacts.splice(index, 1);

            localStorage.setItem('contacts', JSON.stringify(contacts));

            renderContacts();

            notifyExpiredContacts();

        }

    }

    function sortContacts() {

        contacts.sort((a, b) => a.name.localeCompare(b.name));

        renderContacts();

    }

    function filterContacts() {

        const query = document.getElementById('search').value.toLowerCase();

        const filtered = contacts.filter(contact =>

            contact.name.toLowerCase().includes(query) ||

            contact.city.toLowerCase().includes(query)

        );

        basicTableBody.innerHTML = '';

        filtered.forEach((contact, index) => {

            const isAttention = new Date() - new Date(contact.lastContact) > 15 * 24 * 60 * 60 * 1000;

            const tr = document.createElement('tr');

            if (isAttention) {

                tr.classList.add('attention');

            }

            tr.innerHTML = `

                <td><img src="${contact.photo}" alt="Foto" class="contact-photo"></td>

                <td><a href="#" onclick="toggleDetails(${index})">${contact.name}</a></td>

                <td><a href="https://www.google.com/maps/search/${encodeURIComponent(contact.city)}" target="_blank">${contact.city}</a></td>

                <td>

                    <button class="btn btn-warning btn-sm" onclick="editContact(${index})">Editar</button>

                    <button class="btn btn-danger btn-sm" onclick="deleteContact(${index})">Excluir</button>

                </td>

            `;

            const detailsRow = document.createElement('tr');

            detailsRow.classList.add('details-row');

            detailsRow.id = `details-${index}`;

            detailsRow.innerHTML = `

                <td colspan="4">

                    <strong>Telefone:</strong> <a href="https://wa.me/${contact.phone.replace(/\D/g, '')}" target="_blank">${contact.phone}</a><br>

                    <strong>Data de Nascimento:</strong> ${contact.dob}<br>

                    <strong>Grupo:</strong> ${contact.group}<br>

                    <strong>Prioridade:</strong> ${contact.priority}<br>

                    <strong>Último Contato:</strong> ${contact.lastContact}<br>

                    <strong>Notas:</strong> ${contact.notes}<br>

                </td>

            `;

            basicTableBody.appendChild(tr);

            basicTableBody.appendChild(detailsRow);

        });

    }

    function clearFilters() {

        document.getElementById('search').value = '';

        filterContacts();

    }

    function downloadExcel() {

        const wb = XLSX.utils.book_new();

        const ws = XLSX.utils.json_to_sheet(contacts);

        XLSX.utils.book_append_sheet(wb, ws, "Contatos");

        XLSX.writeFile(wb, "contatos.xlsx");

    }

    // Inicializa a renderização e notificação

    renderContacts();

    notifyExpiredContacts();

</script>
